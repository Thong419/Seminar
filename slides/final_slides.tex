\begin{frame}{Business Problem and Motivation}
  \begin{itemize}
    \item Manual verification is slow compared to misinformation spread.
    \item Review teams need fast first-pass triage with evidence support.
    \item Goal: improve review efficiency while keeping decisions explainable.
  \end{itemize}
  \begin{block}{Success targets}
    Business: reduce triage time. Technical: high F1 and stable latency.
  \end{block}
\end{frame}

\begin{frame}{Proposed NLP Solution}
  \begin{columns}[T,onlytextwidth]
    \column{0.52\textwidth}
    \begin{itemize}
      \item Baseline model: TF-IDF + Logistic Regression
      \item Main model: RoBERTa text classifier
      \item Agentic orchestration for evidence and decision synthesis
      \item FastAPI backend + Streamlit frontend
    \end{itemize}
    \column{0.44\textwidth}
    \begin{block}{Output}
      Prediction, confidence, trust score, evidence summary, explanation.
    \end{block}
  \end{columns}
\end{frame}

\begin{frame}{Data Overview}
  \begin{itemize}
    \item Dataset format: CSV with \texttt{text} and \texttt{label}.
    \item Labels: fake and real.
    \item Split strategy: 70\% train, 15\% validation, 15\% test (stratified).
    \item Preprocessing: clean HTML/URL, normalize text, remove duplicates and empty rows.
  \end{itemize}
\end{frame}

\begin{frame}{Deployment Overview}
  \begin{itemize}
    \item Backend deployment: Render free tier (FastAPI service).
    \item Frontend deployment: Streamlit Community Cloud.
    \item Communication via \texttt{API\_URL} environment variable.
    \item Runtime endpoints: \texttt{/health}, \texttt{/version}, \texttt{/predict}, \texttt{/analyze}, \texttt{/agent}.
  \end{itemize}
  \begin{block}{Demo readiness}
    Public URL can be accessed from any machine for course presentation.
  \end{block}
\end{frame}

\begin{frame}{Monitoring and Continual Learning Plan}
  \begin{itemize}
    \item Track confidence drift, prediction drift, uncertain rate, conflict rate, and latency.
    \item Collect reviewed error cases as new training candidates.
    \item Trigger periodic retraining when monitoring thresholds are violated.
  \end{itemize}
\end{frame}

\begin{frame}{Ethics, Privacy and Risks}
  \begin{itemize}
    \item Privacy: PII masking before inference and logging.
    \item Ethics: false positives and false negatives can both cause harm.
    \item Explainability: evidence and trust score improve transparency.
    \item Misuse risk: over-reliance on automated output without human review.
  \end{itemize}
  \begin{block}{Mitigation}
    Use \texttt{UNCERTAIN} state and human escalation for ambiguous cases.
  \end{block}
\end{frame}

\begin{frame}{Key Takeaways and Future Work}
  \begin{itemize}
    \item Built an end-to-end NLP system aligned with production-style requirements.
    \item Combined strong modeling performance with deployability and agentic reasoning.
    \item Next steps: stronger stance modeling, broader robustness tests, and tighter MLOps automation.
  \end{itemize}
\end{frame}
